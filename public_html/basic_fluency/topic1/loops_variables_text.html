//For Loop
for(var i = 0; i <= 10; i++){
document.write(i);
}
document.write("

");

//Nested Loop
for(var i = 0; i <= 10; i++){
for(var x = 0; x <= i; x++){
document.write(x);
}
document.write("
");
}

//While Loop
i = 0;
x = 0;
document.write("
");
while (i < 10) {
document.write(i);
i++;
}

//Do While Loop
i = 0;
x = 0;

document.write("

"); do {
x = i * i;
document.write(x);
document.write("
");
i++;
}
while(x < 100);
